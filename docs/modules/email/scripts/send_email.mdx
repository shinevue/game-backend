## Request

export interface Request {
  from: Email;
  to: Email[];
  cc?: Email[];
  bcc?: Email[];
  replyTo?: Email[];
  subject: string;
  html?: string;
  text?: string;
}
/**
 * Identifies an email address and optionally a name.
 */
export interface Email {
  /**
   * Target email.
   */
  email: string;
  /**
   * Optional display name for the email.
   */
  name?: string;
}
